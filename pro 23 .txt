float CircleAreaByITriangal(float A, float B, float C) {
	const float PI = 3.1415926535;
	float P;
	 P = (A + B + C) / 2;
	float Area = PI * pow(((A * B * C) / (4 * sqrt(P * (P - A) * (P - B) * (P - C)))),2);
	return Area;
}

	cout << "\nCircle area = " << area << endl;
