#include<iostream>
using namespace std;
#include<string>

int ReadNumberInRange(int from,int to) {
	int Grade;
	do {
		cout << "Please enter a grade between 0 and 100 ? " << endl;
		cin >> Grade;
	} while (Grade < from || Grade > to);
	return Grade;
}

char GetGradeLetter(int grade) {
   
	if (grade >= 90)
		return ' A ';
	else if (grade >= 80)
		return 'B';
	else if (grade >= 70)
		return  'C';
	else if (grade >= 60)
		return  'D';
	else if (grade >= 50)
		return 'E';
	else
		return 'F';

}

int main() {
cout<<endl<<"Result = "<<GetGradeLetter(ReadNumberInRange(0, 100));

	return 0;
}
