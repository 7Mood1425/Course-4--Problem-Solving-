#include<iostream>
using namespace std;
#include<string>

float ReadPositiveNumber(string massage) {
	int number = 0;
	do {
		cout << massage;
		cin >> number;

	} while (number <= 0);
	return number;
}
	
float MonthlyInstallment(int loanAmount, int HowManyManth) {
	return (float)loanAmount / HowManyManth;
}
int main() {

	float loanAmount = ReadPositiveNumber("please enter your loan amount :");
	float HowManyManth = ReadPositiveNumber("Please enter How many month : ");
	cout << "Monthly installment = " << MonthlyInstallment(loanAmount, HowManyManth)<<"\n";


	return 0;
}
